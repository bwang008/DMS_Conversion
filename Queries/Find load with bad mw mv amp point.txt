select a.loadname, a.loadnumber, a.busnumber, b.analogpointnumber, c.unitname from loaddefinition a
inner join psa_analogmeasurement b
	on a.amperetelemetryset=b.measurementset 
inner join analogpoint c
	on b.analogpointnumber = c.pointnumber
where b.analogpointnumber < 10
UNION
select a.loadname, a.loadnumber, a.busnumber,b.analogpointnumber, c.unitname from loaddefinition a
inner join psa_analogmeasurement b
	on a.MWtelemetryset=b.measurementset 
inner join analogpoint c
	on b.analogpointnumber = c.pointnumber
where b.analogpointnumber < 10
UNION
select a.loadname, a.loadnumber, a.busnumber,b.analogpointnumber, c.unitname from loaddefinition a
inner join psa_analogmeasurement b
	on a.MVtelemetryset=b.measurementset 
inner join analogpoint c
	on b.analogpointnumber = c.pointnumber
where b.analogpointnumber < 10
order by loadnumber;
